number_1 = int(input("Enter your number1 : ")
number_2 = int(input("Enter your number2 : ")
operation = input(" Enter your operation : ")

print("Results")

if operation == "+":
  print(number_1 + number_2)
if operation == "-":
  print(number_1 - number_2)
if operation == "*":
  print(number_1 * number_2)
if operation == "/":
  print(number_1 / number_2)
if operation == "**":
	print(number_1 ** number_2)
if operation == "//":
	print(number_1 // number_2)
if operation == %:
	print(number_1 % number_2)
